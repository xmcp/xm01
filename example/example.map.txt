// 这是一个简单的示例地图，阅读前请先学习帮助文档 help.txt

// 本地图的相关常数，不是必需的，但这样做可以方便日后修改
% set width=1600
% set height=600

% filter walls
  || 0 0~[height]
  || [width] 0~[height]
  -- 0~[width] 0
  -- 0~[width] [height] @0 // 方框的顶，不允许出生
% endfilter

% filter shops
  % for pos in range(200,width,200)
    [pos] 0 // 每 200 单位宽有一个商店
  % endfor
% endfilter

% filter attributes
  event keep time = -1 // 道具不能卡在顶端的墙上
% endfilter
